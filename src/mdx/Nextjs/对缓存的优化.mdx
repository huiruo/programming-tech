

## 我想 src/pages/game.tsx  和 /central/  也要不要重新执行首页，你怎么改？
真正要让 /game、/central 回首页时首页不重新执行，得把 keepAlive 改成“即使当前路由不是缓存页，也不能把已缓存首页从 React 树里卸载”。

这次改的是根因：以前一旦进入非缓存页，比如 /game、/central，KeepAliveContainer 会直接 return `<Component ... />`，把首页那棵缓存树整个从 React 树里拿掉。现在我改成“缓存页永远留在树上，只是隐藏；非缓存页额外挂一层当前页”。

// 缓存代码
// 缓存代码
import { ReactElement, useState } from "react"
import { usePathname } from "next/navigation"

// 配置需要缓存的路由
export const KEEP_ALIVE_ROUTES = [
  "/",
  "/promotion/",
  "/deposit/",
  "/referral/",
  "/profile/",
]

// 缓存配置
export const CACHE_CONFIG = {
  maxAge: 10 * 60 * 1000,
  enableExpiration: false,
}

interface CachedComponent {
  pathname: string
  component: ReactElement
  timestamp: number
}

interface KeepAliveContainerProps {
  // eslint-disable-next-line @typescript-eslint/no-explicit-any
  Component: any
  // eslint-disable-next-line @typescript-eslint/no-explicit-any
  pageProps: any
  cacheRoutes: string[]
  maxAge?: number
  enableExpiration?: boolean
}

export const KeepAliveContainer = ({
  Component,
  pageProps,
  cacheRoutes,
  maxAge = 10 * 60 * 1000,
  enableExpiration = true,
}: KeepAliveContainerProps) => {
  const [cachedComponents, setCachedComponents] = useState<Map<string, CachedComponent>>(new Map())
  const pathname = usePathname() || ""

  const currentPath = pathname === "/" ? "/" : `${pathname}`

  const shouldCache = (pathname: string) =>
    cacheRoutes.some(route => route.endsWith('*') ? pathname.startsWith(route.slice(0, -1)) : pathname === route)

  const isCacheExpired = (timestamp: number) => enableExpiration && Date.now() - timestamp > maxAge

  // 渲染前同步处理缓存，避免组件双重挂载
  const needsCache = shouldCache(currentPath)
  const existing = cachedComponents.get(currentPath)
  const isExpired = existing && isCacheExpired(existing.timestamp)

  // 如果需要缓存但还没有缓存（或已过期），在渲染时直接创建
  if (needsCache && (!existing || isExpired)) {
    const newCache = new Map(cachedComponents)
    newCache.set(currentPath, {
      pathname: currentPath,
      component: <Component {...pageProps} key={currentPath} />,
      timestamp: Date.now()
    })
    // 只在第一次渲染时同步更新，避免无限循环
    if (cachedComponents !== newCache) {
      // 使用 setTimeout 将状态更新推迟到渲染完成后
      Promise.resolve().then(() => setCachedComponents(newCache))
    }
  } else if (needsCache && existing && !isExpired) {
    // 更新访问时间戳
    if (Date.now() - existing.timestamp > 1000) { // 避免频繁更新
      Promise.resolve().then(() => {
        setCachedComponents(prev => {
          const newCache = new Map(prev)
          const current = newCache.get(currentPath)
          if (current) {
            newCache.set(currentPath, { ...current, timestamp: Date.now() })
          }
          return newCache
        })
      })
    }
  }

  /*
  // 渲染逻辑--改动
  if (!needsCache) {
    // 不需要缓存，直接渲染
    return <Component {...pageProps} key={currentPath} />
  } 
  */

  // 渲染逻辑
  return (
    <>
      {Array.from(cachedComponents.entries()).map(([pathname, cached]) => {
        const isActive = pathname === currentPath
        return (
          <div
            key={pathname}
            style={{
              display: isActive ? "block" : "none",
              height: "100%",
            }}
          >
            {cached.component}
          </div>
        )
      })}

      {!needsCache && (
        <div style={{ height: "100%" }}>
          <Component {...pageProps} key={currentPath} />
        </div>
      )}
    </>
  )
}
